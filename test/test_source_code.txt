


func add(a: int, b: int) {
    let x: int = 10;
    let y: int = 20;
    let result: int = a + b * 2 - 1 / 5;

    if (result >= y) {
        result = result - 1;
    }

    if (result <= y) {
        result = result + 1;
    }

    if (result == y) {
        result = y;
    }

    if (result != y) {
        result = x;
    }

    if (!0) {
        result = 0;
    }

    while (result > x) {
        result = result - 1;
    }

    while (result < y) {
        result = result + 1;
    }

    for (i = 0; i < 10; i = i + 1) {
        result = result + i;
    }

    x = result;
}
